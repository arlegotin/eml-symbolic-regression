```python
MACRO_RULES = (
    "direct_division_template",
    "scaled_exp_minus_one_template",
)

def _macro_diagnostics(source, config, compiled, macro_hits) -> dict[str, Any]:
    return {
        "hits": list(macro_hits),
        "misses": [rule for rule in MACRO_RULES if rule not in macro_hits],
        "baseline_depth": baseline_depth,
        "baseline_node_count": baseline_nodes,
        "depth_delta": baseline_depth - compiled.depth(),
        "node_delta": baseline_nodes - compiled.node_count(),
    }
```
